# Initialize the project
    # If created from scratch
        > npm init -y
    
    # If cloned repository, to install all the dependencies
        > npm install
    
    # Installations
        # Express
            > npm i express
        
        # Nodemon
            > npm i --save-dev nodemon

    # Commands
        > npm run start
